class Round(nn.Module):
    def __init__(self, levels=255):
        super(Round, self).__init__()
        self.gain=(1-1e-6)*levels
    def forward(self, x):
        x = self.gain*x
        if self.training:
            noise = torch.rand_like(x) - 0.5
            x = x + noise
        else:
            x = torch.round(x)
        return x/self.gain
            
class Walloc(nn.Module):
    def __init__(self, in_channels=3, J=4, N=512, latent_dims=[12,12,36,144,576], hyper_dim=64):
        super().__init__()
        self.in_channels = in_channels
        self.J = J
        self.latent_dims = latent_dims
        self.wt  = DTCWTForward(J=J, biort='near_sym_b', qshift='qshift_b')
        self.iwt = DTCWTInverse(biort='near_sym_b', qshift='qshift_b')
        self.encoders = nn.ModuleList()
        self.decoders = nn.ModuleList()
        
        for i, latent_dim in enumerate(latent_dims):
            channels = in_channels if i==0 else 12*in_channels
            rr = 2 if i==0 else i
            self.encoders.append(nn.Sequential(
                autoencoders.autoencoder_kl.Encoder(
                    in_channels = channels,
                    out_channels = latent_dim,
                    down_block_types = rr*('DownEncoderBlock2D',),
                    block_out_channels = rr*(N,),
                    layers_per_block = 2,
                    norm_num_groups = 32,
                    act_fn = 'silu',
                    double_z = False,
                    mid_block_add_attention=True,
                ),
            ))
            self.decoders.append(nn.Sequential(
                autoencoders.autoencoder_kl.Decoder(
                    in_channels = latent_dim,
                    out_channels = channels,
                    up_block_types = rr*('UpDecoderBlock2D',),
                    block_out_channels = rr*(N,),
                    layers_per_block = 2,
                    norm_num_groups = 32,
                    act_fn = 'silu',
                    mid_block_add_attention=True,
                ),
            ))  

        self.hyper_encode = nn.Sequential(
                autoencoders.autoencoder_kl.Encoder(
                    in_channels = sum(latent_dims),
                    out_channels = hyper_dim,
                    down_block_types = ('DownEncoderBlock2D',),
                    block_out_channels = (N,),
                    layers_per_block = 2,
                    norm_num_groups = 32,
                    act_fn = 'silu',
                    double_z = False,
                    mid_block_add_attention=True,
                ),
                torch.nn.Hardtanh(min_val=-128.499, max_val=127.49),
                Round()
            )

        self.hyper_decode = nn.Sequential(
                autoencoders.autoencoder_kl.Decoder(
                    in_channels = hyper_dim,
                    out_channels = sum(latent_dims),
                    up_block_types = ('UpDecoderBlock2D',),
                    block_out_channels = (N,),
                    layers_per_block = 2,
                    norm_num_groups = 32,
                    act_fn = 'silu',
                    mid_block_add_attention=True,
                ),
            )
        
        self.clamp = torch.nn.Hardtanh(min_val=-0.5, max_val=0.5)
        
    def wavelet_analysis(self,x):
        L, H = self.wt(x)
        X = [L] + [einops.rearrange(h, 'b c o h w ri -> b (c o ri) h w') for h in reversed(H)]
        return X

    def encode_components(self,X):
        Y = [self.encoders[i](Xi) for i, Xi in enumerate(X)]
        Y = torch.cat(Y, dim=1)
        return Y

    def decode_components(self,Y_hat):
        Y_hat = torch.split(Y_hat, self.latent_dims, dim=1)
        X_hat = [self.decoders[i](yi) for i, yi in enumerate(Y_hat)]
        return X_hat

    def wavelet_synthesis(self,X_hat):
        L_hat = X_hat[0]
        H_hat = [einops.rearrange(h, 'b (c o ri) h w -> b c o h w ri', c=self.in_channels, o=6, ri=2) for h in reversed(X_hat[1:])]
        x_hat = self.iwt((L_hat, H_hat))
        return x_hat

    def forward(self, x):
        X = self.wavelet_analysis(x)
        Y = self.encode_components(X)
        Z = self.hyper_encode(Y)
        Y_hat = self.hyper_decode(Z)
        X_hat = self.decode_components(Y_hat)
        x_hat = self.wavelet_synthesis(X_hat)
        return self.clamp(x_hat)

def minus_cosine_warmup(i_step):
    scale = 0.5 * (np.log10(args.max_lr) - np.log10(args.min_lr))
    angle =  np.pi * i_step / (args.warmup_steps//args.plot_update)
    log_lr = np.log10(args.min_lr) + scale * (1 - np.cos(angle))
    lr = 10 ** log_lr
    return lr/args.min_lr
    
warmup = torch.optim.lr_scheduler.LambdaLR(
    optimizer,
    lr_lambda = lambda i_step: minus_cosine_warmup(i_step)
)
